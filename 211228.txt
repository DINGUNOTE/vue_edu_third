# axios 가이드
1. HTTP Request & Response와 관련된 기본 설정
const config = {
  baseUrl: 'https://api.hnpwa.com/v0/'
}
2. API 함수들을 정리
function fetchNewsList() {
  // return axios.get(config.baseUrl + 'news/1.json');
  // ES6 템플릿 문법
  return axios.get(`${config.baseUrl}news/1.json`);
}
